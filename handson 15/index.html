Inheritance and Prototype

Problem 1:


Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

Output:-

        // Parent Object
        function Person() {}

        Person.prototype.setName = function(name) {
        this.name = name;
        };

        Person.prototype.getName = function() {
        return this.name;
        };

        // Child Object
        function Student() {}

        Student.prototype = Object.create(Person.prototype);

        // Use the methods from the parent
        const student = new Student();
        student.setName('Ashish Singh');
        console.log(student.getName()); // Output: Ashish Singh

Problem 2:


Write code to explain prototype chaining

Output:-

        // Parent Object
        function Person() {}
        
        Person.prototype.setName = function(name) {
          this.name = name;
        };
        
        Person.prototype.getName = function() {
          return this.name;
        };
        
        // Child Object
        function Student() {}
        
        Student.prototype = Object.create(Person.prototype);
        
        Student.prototype.setAge = function(age) {
          this.age = age;
        };
        
        Student.prototype.getAge = function() {
          return this.age;
        };
        
        // Grandchild Object
        function GraduateStudent() {}
        
        GraduateStudent.prototype = Object.create(Student.prototype);
        
        GraduateStudent.prototype.setDegree = function(degree) {
          this.degree = degree;
        };
        
        GraduateStudent.prototype.getDegree = function() {
          return this.degree;
        };
        
        const gradStudent = new GraduateStudent();
        gradStudent.setName('Ashish');
        gradStudent.setAge(23);
        gradStudent.setDegree('B.tech');
        console.log(gradStudent.getName()); // Output: Ashish
        console.log(gradStudent.getAge()); // Output: 23
        console.log(gradStudent.getDegree()); // Output: B.tech

Problem 3:


Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays

Output:-

      Array.prototype.sum = function() {
        return this.reduce((acc, val) => acc + val, 0);
      };
      
      const array1 = [1, 2, 3, 4, 5];
      const array2 = [10, 20, 30, 40, 50];
      const array3 = [100, 200, 300, 400, 500];
      
      const sum1 = array1.sum();
      const sum2 = array2.sum();
      const sum3 = array3.sum();
      
      console.log(sum1);  // 15
      console.log(sum2);  // 150
      console.log(sum3);  // 1500

Problem 4:


Write a JavaScript function to retrieve all the names of object's own and inherited properties.

Output:-

      function getAllProperties(obj) {
        const properties = [];
        let currentObject = obj;
      
        while (currentObject !== null) {
          properties.push(...Object.getOwnPropertyNames(currentObject));
          currentObject = Object.getPrototypeOf(currentObject);
        }
      
        return properties;
      }
      
      const obj = {
        name: 'John',
        age: 30
      };
      
      console.log(getAllProperties(obj));
      // Output: [ 'name', 'age', 'constructor', 'toString', ... ]
